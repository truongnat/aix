const fs = require("node:fs");
const path = require("node:path");

const storePath = process.env.TODO_FILE || path.join(__dirname, "..", "todos.json");

function loadTodos() {
  try {
    return JSON.parse(fs.readFileSync(storePath, "utf8"));
  } catch (error) {
    if (error.code === "ENOENT") return [];
    throw error;
  }
}

function saveTodos(todos) {
  fs.mkdirSync(path.dirname(storePath), { recursive: true });
  fs.writeFileSync(storePath, JSON.stringify(todos, null, 2) + "\n");
}

function usage() {
  console.log("Usage: todo <add|list|done|remove|clear> [text|number]");
}

function parseIndex(value, todos) {
  const index = Number.parseInt(value, 10);
  if (!Number.isInteger(index) || index < 1 || index > todos.length) {
    throw new Error("Invalid todo number.");
  }
  return index - 1;
}

function main(argv) {
  const [command, ...args] = argv;
  const todos = loadTodos();

  if (command === "add") {
    const text = args.join(" ").trim();
    if (!text) throw new Error("Todo text is required.");
    todos.push({ text, done: false });
    saveTodos(todos);
    console.log(`Added: ${text}`);
    return;
  }

  if (command === "list") {
    if (todos.length === 0) {
      console.log("No todos.");
      return;
    }
    for (const [index, todo] of todos.entries()) {
      const mark = todo.done ? "x" : " ";
      console.log(`${index + 1}. [${mark}] ${todo.text}`);
    }
    return;
  }

  if (command === "done") {
    const index = parseIndex(args[0], todos);
    todos[index].done = true;
    saveTodos(todos);
    console.log(`Done: ${todos[index].text}`);
    return;
  }

  if (command === "remove") {
    const index = parseIndex(args[0], todos);
    const [removed] = todos.splice(index, 1);
    saveTodos(todos);
    console.log(`Removed: ${removed.text}`);
    return;
  }

  if (command === "clear") {
    saveTodos([]);
    console.log("Cleared all todos.");
    return;
  }

  usage();
  process.exitCode = command ? 1 : 0;
}

try {
  main(process.argv.slice(2));
} catch (error) {
  console.error(error.message);
  process.exitCode = 1;
}

const assert = require("node:assert");
const fs = require("node:fs");
const os = require("node:os");
const path = require("node:path");
const { spawnSync } = require("node:child_process");

const cli = path.join(__dirname, "..", "bin", "todo.js");
const tmpDir = fs.mkdtempSync(path.join(os.tmpdir(), "todo-cli-"));
const todoFile = path.join(tmpDir, "todos.json");

function run(args) {
  return spawnSync(process.execPath, [cli, ...args], {
    env: { ...process.env, TODO_FILE: todoFile },
    encoding: "utf8"
  });
}

let result = run(["list"]);
assert.equal(result.status, 0);
assert.match(result.stdout, /No todos/);

result = run(["add", "ship", "demo"]);
assert.equal(result.status, 0);
assert.match(result.stdout, /Added: ship demo/);

result = run(["add", "write", "tests"]);
assert.equal(result.status, 0);

result = run(["list"]);
assert.equal(result.status, 0);
assert.match(result.stdout, /1\. \[ \] ship demo/);
assert.match(result.stdout, /2\. \[ \] write tests/);

result = run(["done", "1"]);
assert.equal(result.status, 0);

result = run(["list"]);
assert.equal(result.status, 0);
assert.match(result.stdout, /1\. \[x\] ship demo/);

result = run(["remove", "2"]);
assert.equal(result.status, 0);
assert.match(result.stdout, /Removed: write tests/);

result = run(["clear"]);
assert.equal(result.status, 0);
assert.deepEqual(JSON.parse(fs.readFileSync(todoFile, "utf8")), []);

console.log("todo cli tests passed");

# Todo CLI Node Demo

A tiny dependency-free Todo CLI generated by the agentic-sdlc workflow `dev/todo-cli-node-demo`.

Run commands from this directory:

```bash
node bin/todo.js add "Write demo"
node bin/todo.js list
node bin/todo.js done 1
node bin/todo.js remove 1
node bin/todo.js clear
npm test
```

Todos are stored in `todos.json` by default. Tests set `TODO_FILE` so they do not mutate your working todo data.
